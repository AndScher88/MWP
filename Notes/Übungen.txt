function tanken() // Aufgabe 1
{
	$liter1 = 40.5;
	$liter2 = 35.7;
	$preis = 1.499;
	$einheit1 = 'Liter';
	$einheit2 = '€';

    $kosten =  ($liter1 + $liter2) * $preis;
    $gesamt = 'Tanken im Monat von ' . ($liter1 + $liter2) . ' ' . $einheit1;
    $gesamt .= ' ' . 'kosten' . ' ' . $kosten . ' ' . $einheit2;
    echo $gesamt;
}

tanken();

echo '<br><br><br>';

function entfernung()  //Aufgabe 2
{
    $entfernung = 350;
    $bahn_km = 0.26;
    $pkw_km = 0.29;
    $pkw_kosten = $pkw_km * $entfernung;
    $bahn_kosten = $bahn_km * $entfernung;

    $aussage = 'Bei einer Entfernung von ' . $entfernung . 'km ';

    if ($pkw_kosten < $bahn_kosten) {
        $vorteil = $bahn_kosten - $pkw_kosten;
	    $vorteil_rel = $vorteil * 100 / $pkw_kosten;
        $aussage .= 'ist die Fahrt mit dem Pkw um ' . $vorteil . '€=' . $vorteil_rel . '% günstiger';
        echo $aussage;
    } else {
	    $vorteil =  $pkw_kosten - $bahn_kosten;
	    $vorteil_rel = $vorteil * 100 / $bahn_kosten;
	    $aussage .= 'ist die Bahnfahrt um ' . $vorteil . '€=' . $vorteil_rel . '% günstiger';
	    echo $aussage;
    }
}
entfernung();

echo '<br><br><br>';


function niederschlagsmengen()  // Aufgabe 3
{
    $niederschlag_2005 = 630;
    $niederschlag_2006 = 610;
    $niederschlag_2007 = 700;
    $niederschlag_mittel = round(($niederschlag_2005 + $niederschlag_2006 + $niederschlag_2007) / 3, 3);

    $aussage = 'Pro m² fielen in Deutschland in den Jahren 2005-2007 durchschnittlich ' . $niederschlag_mittel
        . ' Liter Niederschlag.<br>';

    if ($niederschlag_2005 < $niederschlag_2006 && $niederschlag_2005 < $niederschlag_2007) {
        $unterschied = $niederschlag_mittel - $niederschlag_2005;
        $unterschied_rel = $unterschied * 100 / $niederschlag_mittel;
        $aussage .=  'Das Jahr 2005 war am trockensten und lag dabei um ' . $unterschied . 'l= ' . $unterschied_rel
            . '% unter der mittleren Niederschlagsmenge.';
        echo $aussage;
    }
	if ($niederschlag_2006 < $niederschlag_2005 && $niederschlag_2006 < $niederschlag_2007) {
		$unterschied = $niederschlag_mittel - $niederschlag_2006;
		$unterschied_rel = $unterschied * 100 / $niederschlag_mittel;
		$aussage .=  'Das Jahr 2006 war am trockensten und lag dabei um ' . $unterschied . 'l= ' . $unterschied_rel
            . '% unter der mittleren Niederschlagsmenge.';
		echo $aussage;
	}
	if ($niederschlag_2007 < $niederschlag_2005 && $niederschlag_2007 < $niederschlag_2006) {
		$unterschied = $niederschlag_mittel - $niederschlag_2007;
		$unterschied_rel = $unterschied * 100 / $niederschlag_mittel;
		$aussage .=  'Das Jahr 2007 war am trockensten und lag dabei um ' . $unterschied . 'l= ' . $unterschied_rel
            . '% unter der mittleren Niederschlagsmenge.';
		echo $aussage;
	}
}

niederschlagsmengen();

function gewichtsklasse(float $gewicht)
{
	$aussage = 'Das Gepäckstück wiegt ';
	switch ($gewicht) {
		case $gewicht <= 20:
			$aussage .= $gewicht . 'kg. Es gehört zur Kategorie S (bis 20kg)';
			break;
		case $gewicht <= 30:
			$aussage .= $gewicht . 'kg. Es gehört zur Kategorie M (bis 30kg)';
			break;
		case $gewicht <= 50:
			$aussage .= $gewicht . 'kg. Es gehört zur Kategorie L (bis 50kg)';
			break;
		case $gewicht <= 70:
			$aussage .= $gewicht . 'kg. Es gehört zur Kategorie XL (bis 70kg)';
			break;
		case $gewicht > 70:
			$aussage .= $gewicht . 'kg. Es gehört zur Kategorie als Reisegepäck unzulässig, als Luftfracht versenden (bis 20kg)';
			break;
	}

	echo $aussage;
}

gewichtsklasse(70.1);

echo '<br><br><br>';
